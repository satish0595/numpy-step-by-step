#Draw the grouped bar grapsh for the following Data
import matplotlib.pyplot as plt
year=["2020","2021","2022","2023"]
cse=[50,80,60,55]
it=[35,50,30,40]
ece=[25,30,40,20]
w=0.2
csebar=[i for i in range(len("year"))]
itbar=[ i+w for i in range(len(cse))]
plt.bar(csebar,cse,width=w,color="red",label="CSE")
plt.bar(itbar,it,width=w,color="green",label="IT")
plt.legend()
plt.show()
