#Sub plot for sin(x) and cos(x)
x=np.arange(0,5*np.pi,0.1)
y=np.sin(x)
z=np.cos(x)
#select plot-1
plt.subplot(2,1,1)
plt.plot(x,y,color="red")
plt.xlabel("X-axis")
plt.ylabel("y-axis")
plt.title("Sine Series")
#select plot-2
plt.subplot(2,1,2)
plt.plot(x,z,color="darkblue")
plt.xlabel("X-axis")
plt.ylabel("y-axis")
plt.title("Cosine Series")
plt.suptitle("Sin and Cos Series")
plt.show()
