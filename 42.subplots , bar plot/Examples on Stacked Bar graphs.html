#Drawing Stacked Bar Graph--Horizontal
year=["2019","2020","2021","2022"]
cse=[70,40,60,80]
it=[30,20,25,35]
ece=[20,19,20,18]
mec=[12,15,6,18]
ece_start=[cse[i]+it[i] for i in range(len(year))]
mec_satrt=[cse[i]+it[i]+ece[i] for i in range(len(year))]
plt.barh(year,cse,color="red",label="CSE")
plt.barh(year,it,left=cse,color="orange",label="IT")
plt.barh(year,ece,left=ece_start,color="purple",label="ECE")
plt.barh(year,mec,left=mec_satrt,color="green",label="MECH")
plt.legend()
plt.show()
