import pandas as pd
import matplotlib.pyplot as plt
d={"Sunlight":[0,2,4,6,8],"circum":[1.00,2.50,5.25,10.75,11.25]}
df=pd.DataFrame(d)
print(df)
print("----------------------")
plt.scatter(df["Sunlight"],df["circum"],s=10,c="red",marker="o")
plt.xlabel("Sun Light")
